
拉起镜像
docker pull docker-0.unsee.tech/datagrip/sybase:16.0

docker run -d -t -p 5000:5000 --name=sybase16 docker-0.unsee.tech/datagrip/sybase:16.0

//安装isql

cat /opt/sybase/interfaces


isql -v sybase sa mypassword

isql -S 127.0.0.1 -U sa -P mypassword -D test


//角色
db_owner 和 db_datareader

创建库
CREATE DATABASE test
ON master_data = '/var/docker/sybase/store/my_database.dat' (size = 10MB)
LOG ON master_log = '/var/docker/sybase/store/my_database_log.log' (size = 5MB);


chmod -R 755 /var/docker/sybase/store
chown -R sybase:sybase /var/docker/sybase/store

-- 创建服务器级登录
EXEC sp_addlogin 'aka', 'Apom@321', 'test'

-- 将登录映射到数据库用户
EXEC sp_adduser 'aka', 'Apom@321', 'test'

-- 分配权限
GRANT SELECT ON mytable TO myuser

GRANT SELECT, INSERT, UPDATE ON table_name TO username
